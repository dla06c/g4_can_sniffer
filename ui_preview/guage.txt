  <script>
    const SVG_NS = "http://www.w3.org/2000/svg";
    const CX = 260;
    const CY = 260;

    function polar(cx, cy, r, angleDeg) {
      const a = (angleDeg - 90) * Math.PI / 180;
      return { x: cx + r * Math.cos(a), y: cy + r * Math.sin(a) };
    }

    function describeArc(cx, cy, r, startAngle, endAngle) {
      const start = polar(cx, cy, r, endAngle);
      const end = polar(cx, cy, r, startAngle);
      const largeArc = Math.abs(endAngle - startAngle) <= 180 ? "0" : "1";
      const sweep = endAngle >= startAngle ? "0" : "1";
      return `M ${start.x} ${start.y} A ${r} ${r} 0 ${largeArc} ${sweep} ${end.x} ${end.y}`;
    }

    function addPath(group, d, className, extra = {}) {
      const p = document.createElementNS(SVG_NS, "path");
      p.setAttribute("d", d);
      p.setAttribute("class", className);
      Object.entries(extra).forEach(([k, v]) => p.setAttribute(k, v));
      group.appendChild(p);
      return p;
    }

    function addLine(group, p1, p2, className) {
      const line = document.createElementNS(SVG_NS, "line");
      line.setAttribute("x1", p1.x);
      line.setAttribute("y1", p1.y);
      line.setAttribute("x2", p2.x);
      line.setAttribute("y2", p2.y);
      line.setAttribute("class", className);
      group.appendChild(line);
      return line;
    }

    function addText(group, x, y, text, className) {
      const t = document.createElementNS(SVG_NS, "text");
      t.setAttribute("x", x);
      t.setAttribute("y", y);
      t.setAttribute("class", className);
      t.textContent = text;
      group.appendChild(t);
      return t;
    }

    function tick(group, angle, outerR, innerR, className) {
      addLine(group, polar(CX, CY, outerR, angle), polar(CX, CY, innerR, angle), className);
    }

    function buildSpeedometer() {
      const arcGroup = document.getElementById("speed-blue-arc");
      const tickGroup = document.getElementById("speed-ticks");
      const labelGroup = document.getElementById("speed-labels");
      const fuelGroup = document.getElementById("fuel-arc");
      const fuelTicks = document.getElementById("fuel-ticks");

      // Main blue sweep on the left/top and remaining white sweep on the right.
      addPath(arcGroup, describeArc(CX, CY, 225, 305, 15), "blue-track");
      addPath(arcGroup, describeArc(CX, CY, 225, 15, 128), "white-track", { "stroke-width": 4 });

      const values = [0, 20, 40, 60, 80, 100, 120, 140, 160];
      const startAngle = 235;
      const endAngle = 485;
      values.forEach((v, i) => {
        const angle = startAngle + (endAngle - startAngle) * (v / 160);
        tick(tickGroup, angle, 240, 219, "tick-major");
        const pos = polar(CX, CY, 199, angle);
        addText(labelGroup, pos.x, pos.y, String(v), "label");
      });

      for (let v = 0; v <= 160; v += 5) {
        if (v % 20 === 0) continue;
        const angle = startAngle + (endAngle - startAngle) * (v / 160);
        const normalizedAngle = ((angle % 360) + 360) % 360;
        const cls = angle <= 360 ? "tick-minor blue-tick" : "tick-minor";
        tick(tickGroup, angle, 230, 218, cls);
      }

      // Fuel arc at lower edge.
      addPath(fuelGroup, describeArc(CX, CY, 225, 202, 166), "small-blue-track");
      addPath(fuelGroup, describeArc(CX, CY, 225, 166, 180), "red-track");
      for (let i = 0; i <= 5; i++) {
        const a = 202 + (180 - 202) * (i / 5);
        tick(fuelTicks, a, 235, i === 0 || i === 5 ? 217 : 224, i === 0 || i === 5 ? "tick-major" : "tick-minor blue-tick");
      }

      // Speed needle at around 80 MPH.
      const needleEnd = polar(CX, CY, 118, 15);
      const needle = document.getElementById("speedNeedle");
      needle.setAttribute("x1", CX);
      needle.setAttribute("y1", CY - 20);
      needle.setAttribute("x2", needleEnd.x);
      needle.setAttribute("y2", needleEnd.y);
    }

    function buildTachometer() {
      const leftArc = document.getElementById("rpm-left-arc");
      const topTrack = document.getElementById("rpm-top-track");
      const tickGroup = document.getElementById("rpm-ticks");
      const labelGroup = document.getElementById("rpm-labels");
      const tempGroup = document.getElementById("temp-arc");
      const tempTicks = document.getElementById("temp-ticks");

      addPath(leftArc, describeArc(CX, CY, 225, 305, 355), "teal-track");
      addPath(topTrack, describeArc(CX, CY, 225, 355, 58), "white-track", { "stroke-width": 4 });

      const startAngle = 235;
      const endAngle = 485;
      const max = 10;
      for (let v = 0; v <= max; v++) {
        const angle = startAngle + (endAngle - startAngle) * (v / max);
        tick(tickGroup, angle, 240, 219, "tick-major");
        const pos = polar(CX, CY, 201, angle);
        addText(labelGroup, pos.x, pos.y, String(v), "label");
      }

      for (let v = 0; v <= max; v += .2) {
        const rounded = Math.round(v * 10) / 10;
        if (Math.abs(rounded - Math.round(rounded)) < 0.001) continue;
        const angle = startAngle + (endAngle - startAngle) * (v / max);
        let cls = "tick-minor";
        if (v < 2.4) cls = "tick-minor teal-tick";
        if (v >= 8.0 && v < 8.7) cls = "tick-minor yellow-tick";
        if (v >= 8.7) cls = "tick-minor red-tick";
        tick(tickGroup, angle, 230, 218, cls);
      }

      // Temperature arc at bottom right: white/yellow/red band.
      addPath(tempGroup, describeArc(CX, CY, 225, 170, 188), "white-track", { "stroke-width": 4 });
      addPath(tempGroup, describeArc(CX, CY, 225, 188, 194), "white-track", { "stroke-width": 4 });
      addPath(tempGroup, describeArc(CX, CY, 225, 194, 202), "red-track");
      for (let i = 0; i <= 12; i++) {
        const a = 170 + (202 - 170) * (i / 12);
        tick(tempTicks, a, 235, i === 0 || i === 12 ? 217 : 224, i > 9 ? "tick-minor red-tick" : "tick-minor");
      }

      const rpmNeedle = document.getElementById("rpmNeedle");
      const rpmEnd = polar(CX, CY, 219, 355);
      rpmNeedle.setAttribute("x1", CX);
      rpmNeedle.setAttribute("y1", CY);
      rpmNeedle.setAttribute("x2", rpmEnd.x);
      rpmNeedle.setAttribute("y2", rpmEnd.y);

      const tempNeedle = document.getElementById("tempNeedle");
      const tempEnd = polar(CX, CY, 229, 198);
      tempNeedle.setAttribute("x1", CX);
      tempNeedle.setAttribute("y1", CY);
      tempNeedle.setAttribute("x2", tempEnd.x);
      tempNeedle.setAttribute("y2", tempEnd.y);
    }

    buildSpeedometer();
    buildTachometer();

    // Basic runtime hooks. Call these from your CAN/WebSocket layer later.
    window.setGaugeValues = function setGaugeValues({ mph = 90, rpm = 4.9, tempAngle = 198, fuelAngle = 178 } = {}) {
      document.querySelector("#speedGauge .center-value").textContent = Math.round(mph);

      const speedAngle = 235 + (485 - 235) * (Math.max(0, Math.min(mph, 160)) / 160);
      const speedEnd = polar(CX, CY, 118, speedAngle);
      const speedNeedle = document.getElementById("speedNeedle");
      speedNeedle.setAttribute("x2", speedEnd.x);
      speedNeedle.setAttribute("y2", speedEnd.y);

      const rpmAngle = 235 + (485 - 235) * (Math.max(0, Math.min(rpm, 10)) / 10);
      const rpmEnd = polar(CX, CY, 219, rpmAngle);
      const rpmNeedle = document.getElementById("rpmNeedle");
      rpmNeedle.setAttribute("x2", rpmEnd.x);
      rpmNeedle.setAttribute("y2", rpmEnd.y);

      const tempEnd = polar(CX, CY, 229, tempAngle);
      const tempNeedle = document.getElementById("tempNeedle");
      tempNeedle.setAttribute("x2", tempEnd.x);
      tempNeedle.setAttribute("y2", tempEnd.y);

      const fuelEnd = polar(CX, CY, 229, fuelAngle);
      const fuelNeedle = document.getElementById("fuelNeedle");
      fuelNeedle.setAttribute("x2", fuelEnd.x);
      fuelNeedle.setAttribute("y2", fuelEnd.y);
    };
  </script>
